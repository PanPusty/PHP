SELECT marka,model,cena FROM pojazdy WHERE marka = "BM" ORDER BY cena DESC LIMIT 15;

SELECT (SELECT AVG(cena) FROM pojazdy) AS "srednia_ocena",
(SELECT MAX(cena) FROM pojazdy WHERE model ='meta') AS "Maksymalna_cena";

SELECT
p.marka,
p.model,
p.cena,
k.nazwa,
k.doplata
FROM 
pojazdy p
JOIN kolory k ON
p.kolor = k.id
WHERE
p.model = 'alfa';
